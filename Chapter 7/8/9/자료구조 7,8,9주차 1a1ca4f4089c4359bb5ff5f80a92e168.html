반복적 DFS

  • Iterator 객체로 래핑하기 수월함
  • Q1) Iterator란?
    • 반복자. 원소들을 순회할 수 있는 객체 (내부적으로 포인터가 첫 번째 원소만 가리키고 있음)
    • 기존 리스트를 순회하는 것과는 다르게, 다음 원소가 있는지 검사하여 해당 원소를 꺼낸다. 한 번 검색을 완료한 원소는 없어지고, 다음 원소로 넘어감.
    • 다 꺼낸 후에도, 원소 탐색을 시도하면 에러 발생.
  • 교재 p.78 WikiNodeIterable.java는 iterator를 이용한 반복적 DFS로 DOM 트리를 탐색하고 있음.
private class WikiNodeIterator implements Iterator<Node> {
	Deque<Node> stack;
	
	public WikiNodeIterator(Node node) {
		stack = new ArrayDeque<Node>();
		stack.push(root); //가장 처음 원소만 꺼냄
	}

	@Override
	public boolean hasNext() { //다음 원소가 있는지 검사
		return !stack.isEmpty();
	}

	@Override
	public Node next() {
		if (stack.isEmpty()) { //원소가 없는데 검색할 경우 에러 발생
			throw new NoSuchElementException();
		}

		Node node = stack.pop(); //가장 처음 부분 꺼냄
		List<Node> nodes = new ArrayList<Node>(node.childNodes()); //자식 노드 삽입
		Collections.reverse(nodes);
		for (Node child: nodes) {
			stack.push(child);
		}
		return node; //꺼낸 노드 반환
	}
}

//TermCounter : 검색어와 검색어가 페이지에서 등장하는 횟수를 매핑
public class TermCounter {
	
	private Map<String, Integer> map; //검색어와 등장 횟수 매핑
	private String label; //검색어의 출처가 되는 문서 식별(URL 저장)

	public TermCounter(String label) {
		this.label = label;
		this.map = new HashMap<String, Integer>();
	}

	//term, count 삽입
	public void put(String term, int count) {
		map.put(term, count);
	}

	//term에 해당하는 값 조회
	public Integer get(String term) {
		Integer count = map.get(term);
		return count == null ? 0 : count;
	}

	//term을 발견했을 때 count를 하나씩 늘려줌
	public void incrementTermCount(String term) {
		put(term, get(term) + 1);
	}

	//웹페이지를 인덱싱하는 보조 메서드
	//검색어가 있는지 찾음
	public void processElements(Elements paragraphs) {
		for (Node node: paragraphs) {
			processTree(node);
		}
	}

	//트리에 반복문을 실행
	//텍스트를 포함한 노드 검색
	//텍스트를 추출하여 processText 메서드로 전달
	public void processTree(Node root) {
		for (Node node: new WikiNodeIterable(root)) {
			if (node instanceof TextNode) {
				processText(((TextNode) node).text());
			}
		}
	}

	//찾은 text 처리
	public void processText(String text) {
		String[] array = text.replaceAll("\\pP", " ").toLowerCase().split("\\s+");

		for (int i=0; i<array.length; i++) {
			String term = array[i];
			incrementTermCount(term);
		}
	}
}

public class Index {
	
	private Map<String, Set<TermCounter>> index = 
		new HashMap<String, Set<TermCounter>>();

	//검색어와 연관된 집함에 새로운 TermCounter 객체 추가
	public void add(String term, TermCounter tc) {
		Set<TermCounter> set = get(term);

		if (set == null) {
			set = new HashSet<TermCounter>();
			index.put(term, set);
		}

		set.add(tc);
	}

	public Set<TermCounter> get(String term) {
		return index.get(term);
	}

	public void printIndex() {
		for (String term: keySet()) {
			System.out.println(term);

			Set<TermCounter> tcs = get(term);
			for (TermCounter tc: tcs) {
				Integer count = tc.get(term);
				System.out.println(" " + tc.getLabel() + " " + count);
			}
		}
	}
}

//Entry 클래스 : 키와 값의 컨테이너 p.96
public class Entry implements Map.Entry<K, V> {
	private K key;
	private V value;

	public Entry(K key, V value) {
		this.key = key;
		this.value = value;
	}

	@Override
	public K getKey() {
		return key;
	}

	@Override
	public V getValue() {
		return value;
	}
}
//검색
private Entry findEntry(Object target) {
	for (Entry entry: entries) {
		if (equals(target, entry.getKey())) {
			return entry;
		}
	}
	return null;
}

private boolean equals(Object target, Object obj) {
	if (target == null) {
		return obj == null;
	}
	return target.equals(obj);
}
  • equals 메서드 실행 시간 : 엔트리 개수 n에 의존하지 않으므로 상수 시간
  • findEntry 메서드는 엔트리의 개수 n에 비례. 따라서 선형 시간

public V put(K key, V value) {
	Entry entry = findEntry(key); //선형 시간
	if (entry == null) {
		entries.add(new Entry(key, value));
		return null;
	} else {
		V oldValue = entry.getValue(); //상수 시간
		entry.setValue(value); //상수 시간
		return oldValue;
	}
}

public V get(Object key) {
	Entry entry = findEntry(key); //선형 시간
	if (entry == null) {
		return null;
	}
	return entry.getValue();
}

public V remove(Object key) {
	Entry entry = findEntry(key); //선형 시간
	if (entry == null) {
		return null;
	} else {
		V value = entry.getValue();
		entries.remove(entry);
		return value;
	}
}
